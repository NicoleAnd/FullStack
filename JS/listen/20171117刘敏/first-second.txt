for(var m = 1,n = 2,m+n < 25;m++){
	n += 3;
	console.log(m);
}
第一题
0  m=1,n=2;3<25;1  	 =>1
1  m=2,n=5;7<25;2    =>2
2  m=3,n=8;11<25;3   =>3
3  m=4,n=11;15<25;4  =>4
4  m=5,n=14;19<25;5  =>5
5  m=6,n=17;23<25;6  =>6
6  m=7,n=20;27>25;  条件不成立
